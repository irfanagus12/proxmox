	node.startup = automatic
	node.session.auth.authmethod = CHAP
	node.session.auth.username = ...
	node.session.auth.password = ....
	node.session.timeo.replacement_timeout = 15


cd /etc/iscsi/nodes/<TARGET>/<PORTAL>/   (kalo ada kendala login eror salah jalur IP, hapus IP yang salah)


Buat binding iscsi :
iscsiadm -m iface -I iface0 -o new
iscsiadm -m iface -I iface0 -o update -n iface.net_ifacename -v (nama iface, ex : vmbr1)

Log in dg binding:
iscsiadm -m node -T <TARGET_IQN> -p <TARGET_IP> -I iface0 --login

Check session bindings :
iscsiadm -m session -P 3

Logout iscsi :
iscsiadm --mode node --logoutall=all
